-- Create database
CREATE DATABASE landbank;

-- Create occupation table
CREATE TABLE occupation (
    occ_id VARCHAR(10) NOT NULL PRIMARY KEY,
    occ_type ENUM('Employed', 'Self-employed', 'OFW/Overseas Filipino', 'Retired', 'Farmer/Fisher', 
        'Student/Minor', 'Unemployed', 'Housewife', 'Lawyers/Notary/Independent Legal Professional/Accountant', 
        'Government Official') NOT NULL,
    bus_nature VARCHAR(300) NOT NULL
);

-- Create financial record table
CREATE TABLE financial_record (
    fin_code VARCHAR(10) NOT NULL PRIMARY KEY,
    source_wealth VARCHAR(150) NOT NULL,
    mon_income ENUM('Php 30,000.00 and below', 'Php 30,000.01-50,000.00', 'Php 50,000.01-100,000.00', 'Php 100,000.01-500,000.00', 'Over Php 500,000.00') NOT NULL,
    ann_income ENUM('Php 360,000.00 and below', 'Php 360,000.01-600,000.00', 'Php 600,000.01-1,200,000.00', 'Php 1,200,000.01-6,000,000.00', 'Over Php 6,000,000.00') NOT NULL
);

-- Create employer details table
CREATE TABLE employer_details (
    emp_id VARCHAR(10) NOT NULL PRIMARY KEY,
    tin_id VARCHAR(20) NOT NULL UNIQUE,
    empname VARCHAR(100) NOT NULL,
    emp_address VARCHAR(100) NOT NULL,
    phonefax_no VARCHAR(25) NOT NULL,
    job_title VARCHAR(50) NOT NULL,
    emp_date DATE NOT NULL
);

-- Create customer table
CREATE TABLE customer (
    cust_no VARCHAR(10) NOT NULL PRIMARY KEY,
    custname VARCHAR(50) NOT NULL,
    datebirth DATE NOT NULL,
    nationality VARCHAR(20) NOT NULL,
    citizenship VARCHAR(20) NOT NULL,
    custsex ENUM('Male', 'Female') NOT NULL,
    placebirth VARCHAR(20) NOT NULL,
    civilstatus ENUM('Single', 'Married', 'Widowed', 'Divorced') NOT NULL,
    num_children INT NOT NULL,
    mmaiden_name VARCHAR(50) NOT NULL,
    cust_address VARCHAR(100) NOT NULL,
    email_address VARCHAR(50) NOT NULL UNIQUE,
    contact_no VARCHAR(20) NOT NULL,
    occ_id VARCHAR(10),
    fin_code VARCHAR(10),
    CONSTRAINT fk_customer_occ
        FOREIGN KEY (occ_id)
        REFERENCES occupation (occ_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE,  
    CONSTRAINT fk_customer_fin
        FOREIGN KEY (fin_code)
        REFERENCES financial_record (fin_code)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);

-- Create employment_details table (now correctly references customer)
CREATE TABLE employment_details (
    empd INT AUTO_INCREMENT PRIMARY KEY,
    cust_no VARCHAR(10) NOT NULL UNIQUE,  
    emp_id VARCHAR(10), 
    CONSTRAINT fk_employment_customer
        FOREIGN KEY (cust_no)
        REFERENCES customer(cust_no)
        ON DELETE CASCADE 
        ON UPDATE CASCADE,
    CONSTRAINT fk_employment_employer
        FOREIGN KEY (emp_id)
        REFERENCES employer_details(emp_id)
        ON DELETE SET NULL
        ON UPDATE CASCADE
);

-- Create spouse table
CREATE TABLE spouse (
    cust_no VARCHAR(10) NOT NULL,
    sp_code VARCHAR(5) NOT NULL,
    sp_name VARCHAR(100) NOT NULL,
    sp_datebirth DATE NOT NULL,
    sp_profession VARCHAR(50) NOT NULL, 
    PRIMARY KEY (sp_code),
    CONSTRAINT fk_spouse_customer
        FOREIGN KEY (cust_no)
        REFERENCES customer(cust_no)
        ON DELETE CASCADE 
        ON UPDATE CASCADE
);

-- Create company affiliation table 
CREATE TABLE company_affiliation (
    cust_no VARCHAR(10) NOT NULL,    
    depositor_role VARCHAR(30) NOT NULL,
    dep_compname VARCHAR(100) NOT NULL,
    PRIMARY KEY (cust_no, depositor_role, dep_compname),
    CONSTRAINT fk_company_customer
        FOREIGN KEY (cust_no)
        REFERENCES customer(cust_no)
        ON DELETE CASCADE  
        ON UPDATE CASCADE
);

-- Create bank details table
CREATE TABLE bank_details (
    bank_code VARCHAR(20) PRIMARY KEY NOT NULL,
    bank_name VARCHAR(100) NOT NULL,
    branch VARCHAR(100) NOT NULL
);

-- Create existing bank accounts table
CREATE TABLE existing_bank (
    cust_no VARCHAR(10) NOT NULL,
    bank_code VARCHAR(20) NOT NULL,
    acc_type VARCHAR(30) NOT NULL,
    PRIMARY KEY (cust_no, bank_code, acc_type),
    CONSTRAINT fk_bank_customer
        FOREIGN KEY (cust_no)
        REFERENCES customer(cust_no)
        ON DELETE CASCADE  
        ON UPDATE CASCADE,
    CONSTRAINT fk_bank_details
        FOREIGN KEY (bank_code)
        REFERENCES bank_details(bank_code)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

-- Create public official details table
CREATE TABLE public_official_details (
    gov_int_id VARCHAR(10) PRIMARY KEY NOT NULL,
    gov_int_name VARCHAR(100) NOT NULL,
    official_position VARCHAR(100) NOT NULL,
    branch_orgname VARCHAR(100) NOT NULL
);

-- Create customer-public official relationship table 
CREATE TABLE cust_po_relationship (
    cust_no VARCHAR(10),
    gov_int_id VARCHAR(10),
    relation_desc ENUM('Father', 'Mother', 'Mother-in-Law', 'Father-in-Law', 'Son', 'Daughter', 'Son-in-Law',
    'Daughter-in-Law', 'Brother', 'Sister', 'Brother-in-Law', 'Sister-in-Law', 'Grandfather', 'Grandmother', 'Grandson', 'Granddaughter', 'Uncle',
    'Aunt', 'Nephew', 'Niece', 'Cousin', 'Spouse') NOT NULL,
    PRIMARY KEY (cust_no, gov_int_id, relation_desc),
    CONSTRAINT fk_relationship_customer
        FOREIGN KEY (cust_no)
        REFERENCES customer(cust_no)
        ON DELETE CASCADE 
        ON UPDATE CASCADE,
    CONSTRAINT fk_relationship_official
        FOREIGN KEY (gov_int_id)
        REFERENCES public_official_details(gov_int_id)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

    function addDepositorEntry() {
        const container = document.getElementById('depositor-entries');
        const entries = container.querySelectorAll('.depositor-entry');
        if (entries.length >= 3) return;
        const clone = entries[0].cloneNode(true);
        clone.querySelectorAll('input').forEach(input => input.value = '');
        const removeBtn = document.createElement('button');
        removeBtn.type = 'button';
        removeBtn.textContent = 'Remove';
        removeBtn.className = 'remove-btn';
        removeBtn.onclick = function() { clone.remove(); };
        clone.appendChild(removeBtn);
        container.appendChild(clone);
    }
    // Dynamic Bank Account
    function addBankEntry() {
        const container = document.getElementById('bank-entries');
        const entries = container.querySelectorAll('.bank-entry');
        if (entries.length >= 3) return;
        const clone = entries[0].cloneNode(true);
        clone.querySelectorAll('input').forEach(input => input.value = '');
        const removeBtn = document.createElement('button');
        removeBtn.type = 'button';
        removeBtn.textContent = 'Remove';
        removeBtn.className = 'remove-btn';
        removeBtn.onclick = function() { clone.remove(); };
        clone.appendChild(removeBtn);
        container.appendChild(clone);
    }
    // Dynamic Relationship
    function addRelationshipEntry() {
        const container = document.getElementById('relationship-entries');
        const entries = container.querySelectorAll('.relationship-entry');
        if (entries.length >= 6) return; // 2 rows per entry
        // Clone both rows (name row and details row)
        const nameRow = entries[0].cloneNode(true);
        nameRow.querySelectorAll('input').forEach(input => input.value = '');
        const detailsRow = entries[1].cloneNode(true);
        detailsRow.querySelectorAll('input,select').forEach(input => input.value = '');
        // Remove button
        const removeBtn = document.createElement('button');
        removeBtn.type = 'button';
        removeBtn.textContent = 'Remove';
        removeBtn.className = 'remove-btn';
        removeBtn.onclick = function() { nameRow.remove(); detailsRow.remove(); };
        detailsRow.appendChild(removeBtn);
        container.appendChild(nameRow);
        container.appendChild(detailsRow);
    }
